Object destructuring

                
//ES5
var obj = { first: 'Jane', last: 'Doe' };
var f = obj.first,
    l = obj.last;
//ES6
const obj = { first: 'Jane', last: 'Doe' };
const {first: f, last: l} = obj;
console.log(f, l) //Jane Doe

Array destructuring

                
//ES5
var iterable = ['a', 'b'];
var x = iterable[0],
    y = iterable[1];
//ES6
const iterable = ['a', 'b'];
const [x, y] = iterable;
    // x = 'a'; y = 'b'

Rest destructuring

                
//ES5

//ES6

console.log(b); // 2
console.log(rest); // [3, 4, 5]
